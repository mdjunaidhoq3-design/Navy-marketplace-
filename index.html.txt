<script>
// ---------- NAVIGATION ----------
const pages = ['home','profile','shop','inbox','reviews','rules','rank'];
function showPage(page){
    pages.forEach(p => document.getElementById(p).style.display='none');
    document.getElementById(page).style.display='block';
    if(page==='home') displayItems();
    if(page==='profile') displayProfile();
    if(page==='reviews') displayReviews();
    if(page==='rank') displayRanks();
}

// Hamburger toggle
document.getElementById('hamburger').onclick = function(){
    const nav = document.getElementById('navMenu');
    nav.style.display = nav.style.display==='block'?'none':'block';
}

// ---------- LOCAL STORAGE ----------
let user = JSON.parse(localStorage.getItem('discordUser')) || null;
let items = JSON.parse(localStorage.getItem('items')) || [];
let tickets = JSON.parse(localStorage.getItem('tickets')) || [];
let reviews = JSON.parse(localStorage.getItem('reviews')) || [];
let roles = JSON.parse(localStorage.getItem('roles')) || [];
let shopSlots = JSON.parse(localStorage.getItem('shopSlots')) || {Normal:25, Premium:40, VIP:60, extra:0};

// ---------- DISCORD LOGIN (SIMULATED) ----------
document.getElementById('discordLoginBtn').onclick = function() {
    user = { username: "TestUser", id: "123456789", avatar: "https://cdn.discordapp.com/embed/avatars/0.png", role:'Customer'};
    localStorage.setItem('discordUser', JSON.stringify(user));
    alert("Logged in as " + user.username);
    document.getElementById('discordLoginBtn').style.display='none';
    updateMaxSlots();
    showPage('home');
}

// ---------- RULES ----------
function acceptRules(){
    localStorage.setItem('rulesAccepted','true');
    alert("Rules accepted. You can now use the marketplace.");
    showPage('home');
}

// ---------- PROFILE ----------
function displayProfile(){
    if(!user) return;
    document.getElementById('profileName').innerText = user.username;
    document.getElementById('profileAvatar').src = user.avatar;
    document.getElementById('slotInfo').innerText = `Total Slots: ${getMaxSlots()} (used: ${countUserItems()})`;
}

// ---------- SHOP ----------
function getMaxSlots(){
    if(!user) return 0;
    const type = document.getElementById('shopType').value;
    return shopSlots[type]+shopSlots.extra;
}

function updateMaxSlots(){
    document.getElementById('maxSlots').innerText = getMaxSlots();
    document.getElementById('currentSlots').innerText = countUserItems();
}

function countUserItems(){
    if(!user) return 0;
    return items.filter(i=>i.seller===user.username).length;
}

function postItem(){
    if(!user){ alert("Please login first"); return; }
    const name = document.getElementById('itemName').value;
    const price = parseFloat(document.getElementById('itemPrice').value);
    const img = document.getElementById('itemImage').value;
    const type = document.getElementById('shopType').value;
    if(!name || !price || !img){ alert("Fill all fields"); return; }
    if(countUserItems()>=getMaxSlots()){ alert("You have reached your slot limit!"); return; }

    items.push({name, price, img, seller:user.username, type:type});
    localStorage.setItem('items', JSON.stringify(items));
    displayItems();
    updateMaxSlots();
    alert("Item posted!");
}

function buyExtraSlots(){
    const slots = parseInt(document.getElementById('extraSlots').value);
    if(!slots || slots<=0){ alert("Enter a valid number"); return; }
    shopSlots.extra += slots;
    localStorage.setItem('shopSlots', JSON.stringify(shopSlots));
    updateMaxSlots();
    alert(`Purchased ${slots} extra slots!`);
}

// ---------- DISPLAY ITEMS ----------
function displayItems(){
    const container = document.getElementById('itemsContainer');
    container.innerHTML = '';
    const search = document.getElementById('search').value.toLowerCase();
    let filtered = items.filter(i=>i.name.toLowerCase().includes(search));
    filtered.forEach(i=>{
        let finalPrice = i.price;
        if(user && user.role==='VIP') finalPrice = (i.price*0.95).toFixed(2);
        const div = document.createElement('div');
        div.className = 'item';
	        div.innerHTML = `<img src="${i.img}"><b>${i.name}</b><br>$${finalPrice}<br>Seller: ${i.seller}<br><button class="button" onclick="buyItem('${i.name}')">Buy</button>`;
        container.appendChild(div);
    });
}

// ---------- BUY / TICKETS ----------
function buyItem(itemName){
    const item = items.find(i=>i.name===itemName);
    if(!item){ alert("Item not found"); return; }
    let price = item.price;
    if(user && user.role==='VIP') price = (price*0.95).toFixed(2);
    tickets.push({buyer:user.username, seller:item.seller, item:item.name, price:price});
    localStorage.setItem('tickets', JSON.stringify(tickets));
    alert("Ticket created with seller "+item.seller);
    displayTickets();
}

function displayTickets(){
    const container = document.getElementById('ticketsContainer');
    container.innerHTML = '';
    tickets.forEach(t=>{
        if(t.buyer===user.username || t.seller===user.username){
            const div = document.createElement('div');
            div.innerHTML = `<b>${t.item}</b> - $${t.price}<br>Buyer: ${t.buyer}<br>Seller: ${t.seller}`;
            container.appendChild(div);
        }
    });
}

// ---------- REVIEWS ----------
function postReview(){
    const text = document.getElementById('reviewText').value;
    if(!text){ alert("Write something"); return; }
    reviews.push({user:user.username, text});
    localStorage.setItem('reviews', JSON.stringify(reviews));
    displayReviews();
}

function displayReviews(){
    const container = document.getElementById('reviewsContainer');
    container.innerHTML = '';
    reviews.forEach(r=>{
        const div = document.createElement('div');
        div.style.border='1px solid #ccc'; div.style.padding='5px'; div.style.margin='5px';
        div.innerHTML = `<b>${r.user}</b>: ${r.text}`;
        container.appendChild(div);
    });
}

// ---------- RANKS ----------
function displayRanks(){
    const container = document.getElementById('rolesContainer');
    container.innerHTML = '';
    const roleOrder = ['Owner','Middleman','VIP','Premium','Normal','Customer'];
    roleOrder.forEach(role=>{
        container.innerHTML += `<h4>${role}</h4>`;
        roles.filter(r=>r.role===role).forEach(u=>{
            container.innerHTML += `<p>${u.username}</p>`;
        });
    });
}

// ---------- INIT ----------
if(!localStorage.getItem('rulesAccepted')){
    showPage('rules');
}else{
    showPage('home');
}
if(user){
    document.getElementById('discordLoginBtn').style.display='none';
    updateMaxSlots();
}
